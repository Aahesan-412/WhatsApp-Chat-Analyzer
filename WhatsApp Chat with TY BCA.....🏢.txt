             "ten", "eleven", "twelve", "thirteen", "fourteen", "fifteen", "sixteen", "seventeen", "eighteen", "nineteen" };
        static string[] tens = { "", "", "twenty", "thirty", "forty", "fifty", "sixty", "seventy", "eighty", "ninety" };
        static string ConvertNumberToWords(int number)
        {
            if (number == 0)
                return "zero";

            if (number < 20)
            {
                return units[number];
            }
            else if (number < 100)
            {
                int tensPlace = number / 10;
                int unitsPlace = number % 10;
                if (unitsPlace == 0)
                {
                    return tens[tensPlace];
                }

                return tens[tensPlace] + "-" + units[unitsPlace];
            }
            else
            {
                int hundredsPlace = number / 100;
                int remainder = number % 100;
                if (remainder == 0)
                {
                    return units[hundredsPlace] + " hundred";
                }
                else
                {
                    return units[hundredsPlace] + " hundred and " + ConvertNumberToWords(remainder);
                }
            }
        }
        static void Main(string[] args)

        {
            Console.Write("Enter the number ");
            int number = int.Parse(Console.ReadLine());
            if (number < 0 || number > 999)
            {
                Console.WriteLine("Please enter a number between 0 and 999.");
            }
            else
            {
                Console.WriteLine("The number in words is: " + ConvertNumberToWords(number));
            }
            Console.ReadKey();
        }


    }

class Employee {
    int empId;
    String name;
    double basicSalary;

        this.empId = empId;
        this.name = name;
        this.basicSalary = basicSalary;
    }

    void generateSalarySlip() {
        double hra = 0.2 * basicSalary;  // 20% of basic salary
        double da = 0.1 * basicSalary;   // 10% of basic salary
        double grossSalary = basicSalary + hra + da;

        System.out.println("Employee Name : " + name);
        System.out.println("Basic Salary  : $" + basicSalary);
        System.out.println("HRA (20%)     : $" + hra);
        System.out.println("DA (10%)      : $" + da);
        System.out.println("Gross Salary  : $" + grossSalary);
        System.out.println("===========================\n");
    }
}
